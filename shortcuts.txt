x - delete
t - left toolbar
lshift + a - add
lshift + s - cursor to selected
lctrl + a - apply
lctrl + l - make links
lshift + tab - toggle snap to grid
lctrl - inverse snap to grid
lcrtl + p - parent objects
n - object props
w - selection type
z - rendering modes
	2 - material
	4 - wireframe
	6 - solid
	8 - rendered

g - move
s - scale

	x, y, z - single axis
	lshift + x, y, z - exclude axis

		1, 2, 3 ... - transform transform value

m - move to collection

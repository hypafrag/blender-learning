f3 - search menu
x - delete
f - fill
t - left toolbar
m - merge
lshift + a - add
lshift + s - cursor to selected
lctrl + a - apply
lctrl + l - make links
lshift + tab - toggle snap to grid
lctrl - inverse snap to grid
lctrl + p - parent objects
lctrl + space - maximize area
n - object props
w - selection type
z - rendering modes
	2 - material
	4 - wireframe
	6 - solid
	8 - rendered

g - move
s - scale

	x, y, z - single axis
	lshift + x, y, z - exclude axis

		1, 2, 3 ... - transform transform value

m - move to collection
e - extrude
